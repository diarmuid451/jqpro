tag이름이 select요소로 접근, 값을 가져온다.
    - document.getElementsByTagName('select')[0].value;
    
    - a = document.getElementsByTagName('select');
        x = a[0].value;

name이 person요소로 접근, 값을 가져온다.
    - document.getElementsByName('person')[0].value;
    
    - a = document.getElementsByName('person');
       x = a[0].value;

    b = document.getElementsTagName('div');
